00069     template<int Index, class State>
00070     void subscribe_one(std::shared_ptr<State> state) const {
00071 
00072         typedef typename std::tuple_element<Index, tuple_source_type>::type::value_type source_value_type;
00073 
00074         composite_subscription innercs;
00075 
00076         // when the out observer is unsubscribed all the
00077         // inner subscriptions are unsubscribed as well
00078         state->out.add(innercs);
00079 
00080         auto source = on_exception(
00081             [&](){return state->coordinator.in(std::get<Index>(state->source));},
00082             state->out);
00083         if (source.empty()) {
00084             return;
00085         }
00086 
00087         // this subscribe does not share the observer subscription
00088         // so that when it is unsubscribed the observer can be called
00089         // until the inner subscriptions have finished
00090         auto sink = make_subscriber<source_value_type>(
00091             state->out,
00092             innercs,
00093         // on_next
00094             [state](source_value_type st) {
00095                 auto& values = std::get<Index>(state->pending);
00096                 values.push_back(st);
00097                 if (rxu::apply_to_each(state->pending, rxu::list_not_empty(), rxu::all_values_true())) {
00098                     auto selectedResult = rxu::apply_to_each(state->pending, rxu::extract_list_front(), state->selector);
00099                     state->out.on_next(selectedResult);
00100                 }
00101             },
00102         // on_error
00103             [state](std::exception_ptr e) {
00104                 state->out.on_error(e);
00105             },
00106         // on_completed
00107             [state]() {
00108                 if (--state->pendingCompletions == 0) {
00109                     state->out.on_completed();
00110                 }
00111             }
00112         );
00113         auto selectedSink = on_exception(
00114             [&](){return state->coordinator.out(sink);},
00115             state->out);
00116         if (selectedSink.empty()) {
00117             return;
00118         }
00119         source->subscribe(std::move(selectedSink.get()));
