00070     template<int Index, class State>
00071     void subscribe_one(std::shared_ptr<State> state) const {
00072 
00073         typedef typename std::tuple_element<Index, tuple_source_type>::type::value_type source_value_type;
00074 
00075         composite_subscription innercs;
00076 
00077         // when the out observer is unsubscribed all the
00078         // inner subscriptions are unsubscribed as well
00079         state->out.add(innercs);
00080 
00081         auto source = on_exception(
00082             [&](){return state->coordinator.in(std::get<Index>(state->source));},
00083             state->out);
00084         if (source.empty()) {
00085             return;
00086         }
00087 
00088         // this subscribe does not share the observer subscription
00089         // so that when it is unsubscribed the observer can be called
00090         // until the inner subscriptions have finished
00091         auto sink = make_subscriber<source_value_type>(
00092             state->out,
00093             innercs,
00094         // on_next
00095             [state](source_value_type st) {
00096                 auto& value = std::get<Index>(state->latest);
00097 
00098                 if (value.empty()) {
00099                     ++state->valuesSet;
00100                 }
00101 
00102                 value.reset(st);
00103 
00104                 if (state->valuesSet == sizeof... (ObservableN)) {
00105                     auto values = rxu::surely(state->latest);
00106                     auto selectedResult = rxu::apply(values, state->selector);
00107                     state->out.on_next(selectedResult);
00108                 }
00109             },
00110         // on_error
00111             [state](std::exception_ptr e) {
00112                 state->out.on_error(e);
00113             },
00114         // on_completed
00115             [state]() {
00116                 if (--state->pendingCompletions == 0) {
00117                     state->out.on_completed();
00118                 }
00119             }
00120         );
00121         auto selectedSink = on_exception(
00122             [&](){return state->coordinator.out(sink);},
00123             state->out);
00124         if (selectedSink.empty()) {
00125             return;
