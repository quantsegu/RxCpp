00036     struct nextdetacher
00037     {
00038         ~nextdetacher()
00039         {
00040             trace_activity().on_next_return(*that);
00041             if (do_unsubscribe) {
00042                 that->unsubscribe();
00043             }
00044         }
00045         nextdetacher(const this_type* that)
00046             : that(that)
00047             , do_unsubscribe(true)
00048         {
00049         }
00050         template<class U>
00051         void operator()(U u) {
00052             trace_activity().on_next_enter(*that, u);
00053             try {
00054                 that->destination.on_next(std::move(u));
00055                 do_unsubscribe = false;
00056             } catch(...) {
00057                 auto ex = std::current_exception();
00058                 trace_activity().on_error_enter(*that, ex);
00059                 that->destination.on_error(std::move(ex));
00060                 trace_activity().on_error_return(*that);
00061             }
00062         }
00063         const this_type* that;
00064         bool do_unsubscribe;
00065     };
